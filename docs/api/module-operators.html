A curried function version of the binary + operator.

This works with strings as well as with numbers.

console.log(4 + 7)     // 11
console.log(add(4, 7)) // 11
console.log(add(4)(7)) // 11

const add4 = add(4)
console.log(add4(7))   // 11

A curried function version of the == operator.

The c in ceq is for "coercing". You probably mean to use eq instead.

console.log(3 == '3')    // true
console.log(ceq(3, '3')) // true
console.log(ceq(3)('3')) // true

const ceq3 = ceq(3)
console.log(ceq3('3'))   // true

console.log(3 == 4)      // false
console.log(ceq(3, 4))   // false
console.log(ceq(3)(4))   // false

(static) div(x, y) → {number}

Source:

A curried function version of the / operator.

Note that when used with a single argument, that argument is the dividend (the number that gets divided). If you need the argument to be the divisor (the number by which the dividend is divided), use the flip combinator. See the examples below.

console.log(4 / 7)           // 0.5714285714285714
console.log(div(4, 7))       // 0.5714285714285714
console.log(div(4)(7))       // 0.5714285714285714

const div4 = div(4)
console.log(div4(7))         // 0.5714285714285714

console.log(flip(div)(4, 7)) // 1.75
console.log(flip(div)(4)(7)) // 1.75

const dividedBy4 = flip(div, 4)
console.log(dividedBy4(7))   // 1.75

(static) ge(x, y) → {boolean}

Source:
  • operators/ge.js, line 37

A curried function version of the >= operator.

Note that when used with a single argument, that argument is the one on the left side of the >=. If you need the argument to be the one on the right side, you can either swap the arguments and use le or use the flip combinator on ge. See the examples below.

console.log(4 >= 3)      // true
console.log(ge(4, 3))    // true
console.log(ge(4)(3))    // true

const geFrom4 = ge(4)
// This is the same as x => 4 >= x
console.log(geFrom4(3))  // true

// To do x => x >= 4, there are two choices
const ge4a = flip(ge, 4) // Actual x => x >= 4
const ge4b = le(4)       // x => 4 <= x is the same thing
console.log(ge4a(3))     // false
console.log(ge4b(3))     // false

(static) pow(x, y) → {number}

A curried function version of the ** operator.

Note that when used with a single argument, that argument is the base (the number that is raised to a power). If you need the argument to be the exponent (the power that the base is raised to), use the flip combinator. See the examples below.

console.log(2 ** 5)          // 32
console.log(pow(2, 5))       // 32
console.log(pow(2)(5))       // 32

const powerOf2 = pow(2)
console.log(powerOf2(5))     // 32

console.log(flip(pow)(2, 5)) // 25
console.log(flip(pow)(2)(5)) // 25

const square = flip(pow, 2)
console.log(square(5))       // 25
